  4. #define SW_PIN 13 //スイッチが接続されているピンを定義
  6. int RAN_NUM = 1; //ランダムな数を格納する変数
  7. int sw_state = 0; //スイッチの状態を記憶する変数
  9. void setup() {
  10.   
  11.   //入力ピンに設定
  12.   pinMode( SW_PIN, INPUT_PULLUP );
  13.   
  14.   //出力ピンに設定
  15.   pinMode( 4, OUTPUT );
  16.   pinMode( 5, OUTPUT );
  17.   pinMode( 6, OUTPUT );
  18.   pinMode( 7, OUTPUT );

  22. }
  24. void loop() {
  26.   //SWが押されるまでランダムに出目を表示し続ける
  27.   while(sw_state == 0){
  28.     RAN_NUM = random(1,7); //ランダムに1〜6の出目を生成
  30.     ledOff(); //LEDをすべて消灯
  31.     ledOn(RAN_NUM); //出目に応じたLEDを点灯
  33.     //SWが押されたら、状態を移行する（ループを抜ける）
  34.     if(digitalRead(SW_PIN)==LOW){
  35.       sw_state = 1; //状態を移行
  36.       delay(200);
  37.     }
  38.   }
  40.   //SWが押されるまで決まった出目を表示し続ける
  41.   while(sw_state == 1){
  42.     ledOn(RAN_NUM); //出目に応じたLEDを点灯
  43.     
  44.     //SWが押されたら、状態を移行する（ループを抜ける）
  45.     if(digitalRead(SW_PIN)==LOW){
  46.       sw_state = 0; //状態を移行
  47.       delay(200);
  48.     }
  49.   }
  50. }
